    class RNNCell(nn.Module):
    def __init__(self, input_size, hidden_size):
        super().__init__()
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.input_layer = nn.Linear(input_size, hidden_size)
        self.hidden_layer = nn.Linear(hidden_size, hidden_size, bias=False)

    def init(self, batch_size, device):
        # return the initial state
        return torch.zeros(batch_size, self.hidden_size, device=device)

    def forward(self, incoming, state):
        # flag indicates whether the position is valid. 1 for valid, 0 for invalid.
        output = (self.input_layer(incoming) + self.hidden_layer(state)).tanh()
        new_state = output # stored for next step
        return output, new_state


class LSTMCell(nn.Module):
    def __init__(self, input_size, hidden_size):
        super().__init__()
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.ft_layer = nn.Linear(input_size, hidden_size)
        self.ft_hidden_layer = nn.Linear(hidden_size, hidden_size)
        self.it_layer = nn.Linear(input_size, hidden_size)
        self.it_hidden_layer = nn.Linear(hidden_size, hidden_size)
        self.ot_layer = nn.Linear(input_size, hidden_size)
        self.ot_hidden_layer = nn.Linear(hidden_size, hidden_size)

    def init(self, batch_size, device):
        # return the initial state (which can be a tuple)
        return  (torch.zeros(batch_size, self.hidden_size, device=device), torch.zeros(batch_size, self.hidden_size, device=device))

    def forward(self, incoming, state):
        ft= torch.sigmoid(self.ft_layer(incoming) + self.ft_hidden_layer(state[0]))
        it= torch.sigmoid(self.it_layer(incoming) + self.it_hidden_layer(state[0]))
        c_sim= torch.tanh(self.it_layer(incoming) + self.it_hidden_layer(state[0]))
        new_c= ft*state[1] + it*c_sim
        ot= torch.sigmoid(self.ot_layer(incoming) + self.ot_hidden_layer(state[0]))
        ht= ot*torch.tanh(new_c)
        new_h= ht
        output= ht
        return output, (new_h, new_c)


class GRUCell(nn.Module):
    def __init__(self, input_size, hidden_size):
        super().__init__()
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.z_layer = nn.Linear(input_size, hidden_size)
        self.z_hidden_layer = nn.Linear(hidden_size, hidden_size)
        self.r_layer = nn.Linear(input_size, hidden_size)
        self.r_hidden_layer = nn.Linear(hidden_size, hidden_size)
        self.h_layer = nn.Linear(input_size, hidden_size)
        self.h_hidden_layer = nn.Linear(hidden_size, hidden_size)

    def init(self, batch_size, device):
        # return the initial state
        return torch.zeros(batch_size, self.hidden_size, device=device)

    def forward(self, incoming, state):
        z= torch.sigmoid(self.z_layer(incoming) + self.z_hidden_layer(state))
        r= torch.sigmoid(self.r_layer(incoming) + self.r_hidden_layer(state))
        h_sim= torch.tanh(self.h_layer(incoming) + self.h_hidden_layer(r*state))
        new_h= (1-z)*state + z*h_sim
        output= new_h
        new_state= output
        return output, new_state
